37      /**
38       * Game creates the players and the ball
39       *
40       * @param numberOfPlayers
41       */
42      public Game(int numberOfPlayers) {
43          p1score = 0;
44          p2score = 0;
45          p3score = 0;
46          p4score = 0;
47          this.numberOfPlayers = numberOfPlayers;
48          // Create paddles and set color according to the number of players
49          // left
50          player1 = new Paddle(20, GameCanvas.playAreaHeight / 2 - Paddle.height / 2, false);
51          // right
52          player2 = new Paddle(GameCanvas.playAreaWidth - 30, GameCanvas.playAreaHeight / 2 - Paddle.height / 2, false);
53          // top
54          player3 = new Paddle(GameCanvas.playAreaWidth / 2 - Paddle.height / 2, 20, true);
55          // bottom
56          player4 = new Paddle(GameCanvas.playAreaWidth / 2 - Paddle.height / 2, GameCanvas.playAreaHeight - 30, true);
57  
58          // Set colors according to the number of players
59          if (numberOfPlayers == 1) {
60              player1.setColor(Color.RED);
61              player2.setColor(Color.RED);
62              player3.setColor(Color.RED);
63              player4.setColor(Color.RED);
64          }
65          if (numberOfPlayers == 2) {
66              player1.setColor(Color.RED);
67              player2.setColor(Color.BLUE);
68              player3.setColor(Color.RED);
69              player4.setColor(Color.BLUE);
70          }
71          if (numberOfPlayers == 4) {
72              player1.setColor(Color.RED);
73              player2.setColor(Color.GREEN);
74              player3.setColor(Color.BLUE);
75              player4.setColor(Color.YELLOW);
76          }
77          // Create ball
78          ball = new Ball(GameCanvas.playAreaWidth / 2, GameCanvas.playAreaHeight / 2);
79      }

113     /**
114      * Check scores
115      *
116      * @param numberOfPlayers
117      */
118     public void countScore(int numberOfPlayers) {
119         switch (numberOfPlayers) {
120             case 1:
121                 if (paddleCollisionCheck) {
122                     p1score++;
123                 }
124                 if (leftWallCollision || topWallCollision || rightWallCollision || bottomWallCollision) {
125                     p1score--;
126                 }
127                 break;
128             case 2:
129                 if (leftWallCollision || topWallCollision) {
130                     p2score++;
131                 }
132                 if (rightWallCollision || bottomWallCollision) {
133                     p1score--;
134                 }
135                 break;
136             case 4:
137                 if (leftWallCollision) {
138                     p2score++;
139                     p3score++;
140                     p4score++;
141                 }
142                 if (rightWallCollision) {
143                     p1score++;
144                     p2score++;
145                     p4score++;
146                 }
147                 if (topWallCollision) {
148                     p1score++;
149                     p3score++;
150                     p4score++;
151                 }
152                 if (bottomWallCollision) {
153                     p1score++;
154                     p2score++;
155                     p3score++;
156                 }
157                 break;
158             default:
159                 break;
160         }
161     }

163     /**
164      * Check for wall collisions.
165      */
166     public void wallCollision() {
167         leftWallCollision = false;
168         rightWallCollision = false;
169         topWallCollision = false;
170         bottomWallCollision = false;
171         // Score against player1 (left side)
172         if (ball.getXPosition() <= 0) {
173             ball.setXPosition(GameCanvas.playAreaWidth / 2);
174             ball.setYPosition(GameCanvas.playAreaHeight / 2);
175             leftWallCollision = true;
176             // Score against player 3 (right side)
177         } else if (ball.getXPosition() + ballSize >= GameCanvas.playAreaWidth) {
178             ball.setXPosition(GameCanvas.playAreaWidth / 2);
179             ball.setYPosition(GameCanvas.playAreaHeight / 2);
180             rightWallCollision = true;
181         }
182         // Score against player 2 (top side)
183         if (ball.getYPosition() <= 0) {
184             ball.setXPosition(GameCanvas.playAreaWidth / 2);
185             ball.setYPosition(GameCanvas.playAreaHeight / 2);
186             topWallCollision = true;
187             // Score against player 4 (bottom side)
188         } else if (ball.getYPosition() + ballSize >= GameCanvas.playAreaHeight) {
189             ball.setXPosition(GameCanvas.playAreaWidth / 2);
190             ball.setYPosition(GameCanvas.playAreaHeight / 2);
191             bottomWallCollision = true;
192         }
193     }
194 
195     /**
196      * Check for paddle collisions.
197      */
198     public void paddleCollision() {
199         paddleCollisionCheck = false;
200         // Add some random behavior to the ball as it bounces from the paddle
201         double randomNum1 = (((randomGen.nextDouble()) * 2) - 1) / 5;
202         double randomNum2 = (((randomGen.nextDouble()) * 2) - 1) / 5;
203         if (ball.boundingBox.intersects(player1.boundingBox)) {
204             ball.setXSpeed(ballSpeed + randomNum1);
205             paddleCollisionCheck = true;
206         }
207         if (ball.boundingBox.intersects(player2.boundingBox)) {
208             ball.setXSpeed(-ballSpeed + randomNum1);
209             paddleCollisionCheck = true;
210         }
211         if (ball.boundingBox.intersects(player3.boundingBox)) {
212             ball.setYSpeed(ballSpeed + randomNum2);
213             paddleCollisionCheck = true;
214         }
215         if (ball.boundingBox.intersects(player4.boundingBox)) {
216             ball.setYSpeed(-ballSpeed + randomNum2);
217             paddleCollisionCheck = true;
218         }
219     }
