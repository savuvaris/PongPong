5   import javax.swing.JButton;
6   import javax.swing.JFrame;
7   import game.GameCanvas;
8   
9   public class StartMenu extends JFrame {
10  
11      public static int numberOfPlayers;
12  
13      int screenWidth = 250;
14      int screenHeight = 250;
15      int buttonWidth = 150;
16      int buttonHeight = 40;
17  
18      JButton onePlayer, twoPlayer, fourPlayer;
19  
20      public StartMenu() {
21          getContentPane().setLayout(null);
22  
23          // Add buttons
24          onePlayer = new JButton("One Player");
25          getContentPane().add(onePlayer);
26          onePlayer.setBounds((screenWidth - buttonWidth) / 2, 5 + buttonHeight, buttonWidth, buttonHeight);
27  
28          twoPlayer = new JButton("Two Players");
29          getContentPane().add(twoPlayer);
30          twoPlayer.setBounds((screenWidth - buttonWidth) / 2, 2 * (5 + buttonHeight), buttonWidth, buttonHeight);
31  
32          fourPlayer = new JButton("Four Players");
33          getContentPane().add(fourPlayer);
34          fourPlayer.setBounds((screenWidth - buttonWidth) / 2, 3 * (5 + buttonHeight), buttonWidth, buttonHeight);
35  
36          // 
37          pack();
38          setVisible(true);
39          setLocationRelativeTo(null);
40          setSize(screenWidth, screenHeight);
41          setTitle("PongPong");
42          setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
43          setResizable(false);
44  
45          // Actionlisteners
46          onePlayer.addActionListener(new ActionListener() {
47              @Override
48              public void actionPerformed(ActionEvent e) {
49                  dispose();
50                  numberOfPlayers = 1;
51                  GameCanvas pongpong = new GameCanvas();
52                  pongpong.start();
53              }
54          });
55  
56          twoPlayer.addActionListener(new ActionListener() {
57              @Override
58              public void actionPerformed(ActionEvent e) {
59                  dispose();
60                  numberOfPlayers = 2;
61                  GameCanvas pongpong = new GameCanvas();
62                  pongpong.start();
63              }
64          });
65  
66          fourPlayer.addActionListener(new ActionListener() {
67              @Override
68              public void actionPerformed(ActionEvent e) {
69                  dispose();
70                  numberOfPlayers = 4;
71                  GameCanvas pongpong = new GameCanvas();
72                  pongpong.start();
73              }
74          });
75  
76      }
77  
78      public static void main(String[] args) {
79          new StartMenu();
80      }
